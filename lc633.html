                                                bool judgeSquareSum(int c) {
    if (c == 0 || c == 1)
        return true;
    int pos = 0;
    for (int i = 16383; i >= 0; --i) {
        long long temp = (long long)pos * pos + (long long)i * i;
        while (temp <= c) {
            pos++;
            if (temp == c) {
                return true;
            }
            temp = (long long)pos * pos + (long long)i * i;
        }
        
    }
    return false;
}
