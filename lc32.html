                                                class Solution {
    public int longestValidParentheses(String s) {
        int len = s.length();
        // ç®€å•çš„stackï¼Œæé«˜æ•ˆç‡
        int[] stack = new int[len];
        char[] ch = s.toCharArray();
        int top = -1;
        int res = 0, temp = 0;
        // è®°å½•åç‚¹
        boolean[] vis = new boolean[len];
        // éå†åç‚¹
        for(int i = 0; i < len; ++i) {
            if(ch[i] == '(')
                stack[++top] = i;
            else if(ch[i] == ')') {
                if(top != -1) {
                    --top;
                } else {
                    vis[i] = true;
                }
            }
        }
        for(int i = 0; i <= top; ++i)
            vis[stack[i]] = true;
        
        //æ ¹æ®åç‚¹æ¥æ‰¾æœ€å¤§çš„è¿ç»­å€¼
        for(int i = 0; i < len; ++i) {
            if(!vis[i])
                temp++;
            else {
                res = Math.max(res, temp);
                temp = 0;
            }
        }
        // æœ€åä¸€æ¬¡å¯¹æ¯”
        res = Math.max(res, temp);

        return res;
    }
