int cmpfunc (const void * a, const void * b) {
    int *_a = *(int **)a;
    int *_b = *(int **)b;
    if (_a[0] == _b[0]) {
        return _a[1] - _b[1];
    } else {
        return _a[0] - _b[0];
    }
}
